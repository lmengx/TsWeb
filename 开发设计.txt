    player_daily_stat 玩家每日时长（主键防重复，按天聚合）
        CREATE TABLE player_daily_stat (
        uid TEXT NOT NULL,        -- 玩家唯一ID
        date TEXT NOT NULL,       -- 日期 yyyy-MM-dd
        daily_min INTEGER NOT NULL DEFAULT 0, -- 当日游玩分钟
        PRIMARY KEY (uid, date)
    );
    -- 索引：查单个玩家所有天数记录
    CREATE INDEX idx_pds_uid ON player_daily_stat(uid);


    hourly_online_snapshot 每小时在线人数
        CREATE TABLE hourly_online_snapshot (
        hour_ts INTEGER PRIMARY KEY, -- 小时时间戳（整点，如2026062514）
        online_count TEXT NOT NULL -- 该小时在线过的所有玩家的数字id，空格分开
    );
    CREATE INDEX idx_hour_ts ON hourly_online_snapshot(hour_ts);
